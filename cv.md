static getCount(str: string): number {
  let vowelsCount: number = 0;
  for (let i: number = 0; i < str.length; i++){
    if (str[i] === "a" || str[i] === "e" || str[i] === "i" || str[i] === "o" || str[i] === "u") {
      vowelsCount += 1;
    }
  }   
  return vowelsCount;
  }
